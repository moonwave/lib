$ sudo mkdir -p /usr/local/systemc-2.3.3/
$ cd ~/Downloads/systemc-2.3.3
$ mkdir objdir
$ cd objdir

In current objdir do this command:

$ sudo ../configure --prefix=/usr/local/systemc-2.3.3/
$ sudo make
$ sudo make install

$ cd /usr/local
$ sudo ln -s systemc-2.3.3 systemc

# Prerequisite Tools
1. cmake (3.0+)
2. g++ (5.4+)
3. build-essential
4. gdb
5. doxygen
6. graphviz

$ sudo apt update
$ sudo apt install cmake g++ build-essential gdb doxygen graphviz

# Debug build commands

$ cd ~/Downloads/systemc-2.3.3
$ mkdir debug
$ cd debug
$ cmake \
    -DCMAKE_BUILD_TYPE=Debug \
    -DCMAKE_DEBUG_POSTFIX="-d" \
    -DCMAKE_INSTALL_PREFIX=/usr/local/systemc-2.3.3 \
    -DCMAKE_CXX_STANDARD=11 \
    -DBUILD_SOURCE_DOCUMENTATION=ON \
    ..

$ make -j 4
$ make check -j 4
$ sudo make install

# Release build commands #

$ cd ~/Downloads/systemc-2.3.3
$ mkdir release
$ cd release
$ cmake \
    -DCMAKE_BUILD_TYPE=Release \
    -DCMAKE_INSTALL_PREFIX=/usr/local/systemc-2.3.3 \
    -DCMAKE_CXX_STANDARD=11 \
    -DBUILD_SOURCE_DOCUMENTATION=ON \
    ..

$ make -j 4
$ make check -j 4
$ sudo make install

$ cd /usr/local
$ sudo ln -s systemc-2.3.3 systemc

open ~/.bashrc
export SYSTEMC_HOME=/usr/local/systemc
